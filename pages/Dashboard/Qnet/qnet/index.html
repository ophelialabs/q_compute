        // Build network graph based on proximity and type priorities
        function buildNetworkGraph(maxDistanceKm = 8000) {
            const graph = {};
            
            // Initialize graph
            NETWORK_NODES.forEach(node => {
                graph[node.id] = [];
            });

            // Create connections based on distance and prioritization
            NETWORK_NODES.forEach(node => {
                const connections = NETWORK_NODES
                    .filter(n => n.id !== node.id)
                    .map(n => ({
                        id: n.id,
                        node: n,
                        distance: calculateDistance(
                            node.latitude, node.longitude,
                            n.latitude, n.longitude
                        ),
                        priority: n.type === 'teleport' ? 3 : 
                                 (n.type === 'ground-station' || n.type === 'array') ? 2 : 1
                    }))
                    .filter(c => c.distance <= maxDistanceKm)
                    .sort((a, b) => {
                        if (b.priority !== a.priority) return b.priority - a.priority;
                        return a.distance - b.distance;
                    })
                    .slice(0, 8); // Each node connects to up to 8 nearest priority nodes

                connections.forEach(c => {
                    graph[node.id].push(c.id);
                });
            });

            return graph;
        }

        // BFS to find shortest path distances
        function findHopDistances(startNodeId, graph, maxHops = 3) {
            const distances = {};
            const queue = [];

            NETWORK_NODES.forEach(node => {
                distances[node.id] = Infinity;
            });

            distances[startNodeId] = 0;
            queue.push(startNodeId);

            let head = 0;
            while (head < queue.length) {
                const currentId = queue[head];
                head++;

                if (distances[currentId] >= maxHops) continue;

                graph[currentId].forEach(neighborId => {
                    if (distances[neighborId] > distances[currentId] + 1) {
                        distances[neighborId] = distances[currentId] + 1;
                        queue.push(neighborId);
                    }
                });
            }

            return distances;
        }

        // Draw communication lines with hop-distance layering

            clearCommunicationLines();

            // Find hop distances from selected node
            const hopDistances = findHopDistances(selectedNode.id, graph);

            // Define styles for each hop

                1: { color: '#10b981', opacity: 1.0, weight: 2.5, dash: null, label: '1-Hop' },

            // Draw polylines for all reachable nodes within 3 hops

                if (targetNode.id === selectedNode.id) return;

                    title: `${hop}-hop to ${targetNode.name}`
                });

                polylines.push(polyline);
            });

            // Update sidebar with hop statistics
            updateHopStatistics(selectedNode, hopDistances);
        }

        // Calculate distance between two coordinates using Haversine formula
        function calculateDistance(lat1, lon1, lat2, lon2) {
            const R = 6371; // Earth's radius in km
            const dLat = (lat2 - lat1) * Math.PI / 180;
            const dLon = (lon2 - lon1) * Math.PI / 180;
            const a = Math.sin(dLat / 2) * Math.sin(dLat / 2) +
                Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *
                Math.sin(dLon / 2) * Math.sin(dLon / 2);
            const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
            return R * c;
        }

        // Draw communication lines with hop-distance layering
        function drawCommunicationLinesLegacy(selectedNode) {
            clearCommunicationLines();

            // Find the 5 nearest nodes, prioritizing teleport stations and ground stations
            const distances = NETWORK_NODES
                .filter(n => n.id !== selectedNode.id)
                .map(node => ({
                    node: node,
                    distance: calculateDistance(
                        selectedNode.latitude, selectedNode.longitude,
                        node.latitude, node.longitude
                    ),
                    // Priority score: teleport=3, ground-station=2, array=2, node=1
                    priority: node.type === 'teleport' ? 3 : 
                             (node.type === 'ground-station' || node.type === 'array') ? 2 : 1
                }))
                .sort((a, b) => {
                    // Sort by priority first (descending), then by distance (ascending)
                    if (b.priority !== a.priority) return b.priority - a.priority;
                    return a.distance - b.distance;
                })
                .slice(0, 5);

            // Draw polylines to the nearest/prioritized nodes
            distances.forEach(item => {
                const polyline = new google.maps.Polyline({
                    path: [
                        { lat: selectedNode.latitude, lng: selectedNode.longitude },
                        { lat: item.node.latitude, lng: item.node.longitude }

                    strokeColor: item.node.type === 'teleport' ? '#ec4899' :

                polylines.push(polyline);
            });
        }
